29  	private int counter1;

35  	 * Instantiates a new view frame.
36  	 */
37  	public ViewFrame() {

42  	/**
43  	 * Builds the view frame.
44  	 *
45  	 */
46  	private void buildViewFrame() {
47          this.choosePan();
48  		this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
49  		this.setTitle("BoulderDash");
50  		//this.setContentPane(new ViewPanel());
51  		this.setResizable(false);
52  	   // this.addKeyListener(new KeyBoard());
53  		this.setSize(IView.WIDTH,IView.HEIGHT);
54  		this.setLocationRelativeTo(null);
55  		this.setVisible(true);
56  	}
57  
58  	/**
59  	 * 
60  	 * @param pan
61  	 *       The new Panel
62  	 */
63  	public void setContentPanel(JPanel pan) {
64  		this.setContentPane(pan);
65  		//System.out.println("a");
66  		repaint();
67  		revalidate();
68  		counter1=0;
69  	}
70  	
71  	public void choosePan() {
72  	Timer timer=new Timer();	
73  	timer.schedule(new TimerTask(){ 
74  		public void run(){
75  		if(IView.counter%2==0) {
76  	         
77  			controlPanel = new ControlPanel();
78  			setContentPanel(controlPanel);
79  			/* Music */
80  			Music.FirstPlay("music/game.wav");
81  		}
82  		
83  		else if(IView.counter%2==1) {
84  			if(counter1==0) {
85  			viewPanel = new ViewPanel();
86  			setContentPanel(viewPanel);
87  			viewPanel.addKeyListener(new KeyBoard());
88  			viewPanel.setFocusable(true);
89  			viewPanel.requestFocusInWindow();
90  			revalidate();
91  			repaint();
92  			counter1++;
93  			}
94  		}			
95        }
96  	},0, 500);
97    }
